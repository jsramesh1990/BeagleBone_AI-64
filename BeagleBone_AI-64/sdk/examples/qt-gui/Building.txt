# Source SDK environment
source sdk/environment-setup.sh

# Build with qmake
qmake
make

# Or build with cmake
mkdir build && cd build
cmake -DCMAKE_TOOLCHAIN_FILE=../../sdk/cmake/Toolchain.cmake ..
make

# Deploy
scp dashboard root@<board-ip>:/usr/bin/

# Run on target
ssh root@<board-ip>
export DISPLAY=:0
dashboard
